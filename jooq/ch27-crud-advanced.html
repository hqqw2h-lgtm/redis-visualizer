27.1.2 条件拼装

图 27-1.2 · 动态条件拼装 — 4 个 null 检查 + 全部合并 + AND 连接
@Service
@RequiredArgsConstructor
public class OrderQueryService {
    private final DSLContext dsl;

    public List<OrderDto> searchOrders(OrderSearchRequest req) {
        var conditions = buildConditions(req);

        return dsl.select(ORDERS.ID, ORDERS.STATUS, ORDERS.TOTAL_AMOUNT,
                          ORDERS.CREATED_AT, CUSTOMERS.NAME)
            .from(ORDERS)
            .join(CUSTOMERS).on(ORDERS.CUSTOMER_ID.eq(CUSTOMERS.ID))
            .where(conditions)
            .orderBy(resolveSortField(req))
            .limit(req.pageSize()).offset(req.page() * req.pageSize())
            .fetchInto(OrderDto.class);
    }

    private List<Condition> buildConditions(OrderSearchRequest req) {
        var conditions = new ArrayList<Condition>();
        if (req.status() != null)     conditions.add(ORDERS.STATUS.eq(req.status()));
        if (req.customerId() != null) conditions.add(ORDERS.CUSTOMER_ID.eq(req.customerId()));
        if (req.minAmount() != null)  conditions.add(ORDERS.TOTAL_AMOUNT.ge(req.minAmount()));
        if (req.maxAmount() != null)  conditions.add(ORDERS.TOTAL_AMOUNT.le(req.maxAmount()));
        if (req.dateFrom() != null)   conditions.add(ORDERS.CREATED_AT.ge(req.dateFrom()));
        if (req.dateTo() != null)     conditions.add(ORDERS.CREATED_AT.lt(req.dateTo()));
        return conditions;
    }
}

27.1.4 总数 + 数据分页

public Page<OrderDto> searchOrdersPaged(OrderSearchRequest req) {
    var conditions = buildConditions(req);

    int total = dsl.selectCount()
        .from(ORDERS)
        .join(CUSTOMERS).on(ORDERS.CUSTOMER_ID.eq(CUSTOMERS.ID))
        .where(conditions)
        .fetchOneInto(int.class);

    var data = dsl.select(ORDERS.ID, ORDERS.STATUS, ORDERS.TOTAL_AMOUNT,
                          ORDERS.CREATED_AT, CUSTOMERS.NAME)
        .from(ORDERS)
        .join(CUSTOMERS).on(ORDERS.CUSTOMER_ID.eq(CUSTOMERS.ID))
        .where(conditions)
        .orderBy(resolveSortField(req))
        .limit(req.pageSize()).offset(req.page() * req.pageSize())
        .fetchInto(OrderDto.class);

    return new Page<>(data, total, req.page(), req.pageSize());
}

public List<MonthlyStat> getMonthlySales(int year) {
    return dsl.select(
        DSL.extract(ORDERS.CREATED_AT, DatePart.MONTH).as("month"),
        DSL.count().as("order_count"),
        DSL.sum(ORDERS.TOTAL_AMOUNT).as("total_amount"),
        DSL.avg(ORDERS.TOTAL_AMOUNT).as("avg_order_value"),
        DSL.countDistinct(ORDERS.CUSTOMER_ID).as("unique_customers"))
    .from(ORDERS)
    .where(DSL.extract(ORDERS.CREATED_AT, DatePart.YEAR).eq(year))
    .and(ORDERS.STATUS.ne("CANCELLED"))
    .groupBy(DSL.extract(ORDERS.CREATED_AT, DatePart.MONTH))
    .orderBy(DSL.extract(ORDERS.CREATED_AT, DatePart.MONTH))
    .fetchInto(MonthlyStat.class);
}

       COUNT(*)                              AS "order_count",

       COUNT(DISTINCT orders.customer_id)    AS "unique_customers"
FROM orders
WHERE EXTRACT(YEAR FROM orders.created_at) = ?
  AND orders.status <> ?
GROUP BY EXTRACT(MONTH FROM orders.created_at)
ORDER BY EXTRACT(MONTH FROM orders.created_at)

27.2.2 分类销售占比

public List<CategorySales> getCategorySales(int year) {
    var totalSub = dsl.select(DSL.sum(ORDER_ITEMS.UNIT_PRICE.mul(ORDER_ITEMS.QUANTITY)))
        .from(ORDER_ITEMS)
        .join(ORDERS).on(ORDER_ITEMS.ORDER_ID.eq(ORDERS.ID))
        .where(DSL.extract(ORDERS.CREATED_AT, DatePart.YEAR).eq(year))
        .and(ORDERS.STATUS.ne("CANCELLED"))
        .asField("grand_total");

    return dsl.select(
        CATEGORIES.NAME.as("category"),
        DSL.sum(ORDER_ITEMS.UNIT_PRICE.mul(ORDER_ITEMS.QUANTITY)).as("revenue"),
        DSL.count(DSL.distinct(ORDERS.ID)).as("order_count"),
        DSL.round(DSL.sum(ORDER_ITEMS.UNIT_PRICE.mul(ORDER_ITEMS.QUANTITY))
            .mul(100).div(totalSub), 2).as("percentage"))
    .from(CATEGORIES)
    .join(PRODUCTS).on(PRODUCTS.CATEGORY_ID.eq(CATEGORIES.ID))
    .join(ORDER_ITEMS).on(ORDER_ITEMS.PRODUCT_ID.eq(PRODUCTS.ID))
    .join(ORDERS).on(ORDER_ITEMS.ORDER_ID.eq(ORDERS.ID))
    .where(DSL.extract(ORDERS.CREATED_AT, DatePart.YEAR).eq(year))
    .and(ORDERS.STATUS.ne("CANCELLED"))
    .groupBy(CATEGORIES.ID, CATEGORIES.NAME)
    .orderBy(field("revenue").desc())
    .fetchInto(CategorySales.class);
}

public List<CustomerRank> getCustomerRanking() {

        DSL.sum(ORDERS.TOTAL_AMOUNT).as("total_spent"),

27.3.2 7 日移动平均

public List<DailyRevenue> getDailyRevenueWithMovingAvg() {
    return dsl.select(
        DSL.trunc(ORDERS.CREATED_AT).as("date"),
        DSL.sum(ORDERS.TOTAL_AMOUNT).as("daily_revenue"),
        DSL.avg(DSL.sum(ORDERS.TOTAL_AMOUNT))
           .over(orderBy(DSL.trunc(ORDERS.CREATED_AT))
                 .rowsBetweenPreceding(6).andCurrentRow())
           .as("moving_avg_7d"))
    .from(ORDERS)
    .where(ORDERS.STATUS.ne("CANCELLED"))
    .groupBy(DSL.trunc(ORDERS.CREATED_AT))
    .orderBy(field("date"))
    .fetchInto(DailyRevenue.class);
}

public List<MonthlyRunning> getRunningRevenue(int year) {

27.4.1 分类树查询

public List<CategoryTree> getCategoryTree(Long rootId) {
    var tree = name("category_tree")
        .fields("id", "name", "parent_id", "level", "path")
        .as(
        select(CATEGORIES.ID, CATEGORIES.NAME, CATEGORIES.PARENT_ID,
               val(0).as("level"),
               CATEGORIES.NAME.cast(String.class).as("path"))
        .from(CATEGORIES)
        .where(rootId != null
            ? CATEGORIES.ID.eq(rootId)
            : CATEGORIES.PARENT_ID.isNull())
        .unionAll(
            select(CATEGORIES.ID, CATEGORIES.NAME, CATEGORIES.PARENT_ID,
                   field(name("category_tree", "level"), Integer.class).plus(1),
                   DSL.concat(
                       field(name("category_tree", "path"), String.class),
                       val(" > "), CATEGORIES.NAME))
            .from(CATEGORIES)
            .join(table(name("category_tree")))
            .on(CATEGORIES.PARENT_ID.eq(field(name("category_tree", "id"), Long.class)))
            .where(field(name("category_tree", "level"), Integer.class).lt(10))));

    return dsl.withRecursive(tree)
        .selectFrom(tree)
        .orderBy(field(name("path")))
        .fetchInto(CategoryTree.class);
}

WITH RECURSIVE category_tree AS (
    SELECT c.id, c.name, c.parent_id, 0 AS "level",
           CAST(c.name AS VARCHAR) AS "path"
    FROM categories c WHERE c.parent_id IS NULL
    UNION ALL
    SELECT c.id, c.name, c.parent_id, ct."level" + 1,
           ct."path" || ' > ' || c.name
    FROM categories c
    JOIN category_tree ct ON c.parent_id = ct.id
    WHERE ct."level" < 10
)
SELECT * FROM category_tree ORDER BY "path"

public record OrderItemDto(String productName, int quantity, BigDecimal unitPrice) {}
public record OrderSummaryDto(Long id, String status, BigDecimal total,
                              LocalDateTime createdAt, List<OrderItemDto> items) {}
public record CustomerDetail(Long id, String name, String email,
                             List<OrderSummaryDto> recentOrders) {}

public CustomerDetail getCustomerDetail(Long customerId) {
    return dsl.select(
        CUSTOMERS.ID, CUSTOMERS.NAME, CUSTOMERS.EMAIL,
        multiset(
            select(ORDERS.ID, ORDERS.STATUS, ORDERS.TOTAL_AMOUNT, ORDERS.CREATED_AT,
                multiset(
                    select(PRODUCTS.NAME, ORDER_ITEMS.QUANTITY, ORDER_ITEMS.UNIT_PRICE)
                    .from(ORDER_ITEMS)
                    .join(PRODUCTS).on(ORDER_ITEMS.PRODUCT_ID.eq(PRODUCTS.ID))
                    .where(ORDER_ITEMS.ORDER_ID.eq(ORDERS.ID))
                ).as("items").convertFrom(r -> r.into(OrderItemDto.class)))
            .from(ORDERS)
            .where(ORDERS.CUSTOMER_ID.eq(CUSTOMERS.ID))
            .orderBy(ORDERS.CREATED_AT.desc())
            .limit(5)
        ).as("recent_orders").convertFrom(r -> r.into(OrderSummaryDto.class)))
    .from(CUSTOMERS)
    .where(CUSTOMERS.ID.eq(customerId))
    .fetchOne(Records.mapping(CustomerDetail::new));
}

27.6.1 全文搜索

public List<ProductDto> fullTextSearch(String keyword) {
    return dsl.select(PRODUCTS.NAME, PRODUCTS.PRICE)
        .from(PRODUCTS)
        .where(condition(
            "description_tsv @@ plainto_tsquery('simple', {0})", keyword))
        .orderBy(field(
            "ts_rank(description_tsv, plainto_tsquery('simple', {0}))",
            Double.class, keyword).desc())
        .limit(20)
        .fetchInto(ProductDto.class);
}

SELECT products.name, products.price
FROM products
WHERE description_tsv @@ plainto_tsquery('simple', ?)
ORDER BY ts_rank(description_tsv, plainto_tsquery('simple', ?)) DESC
LIMIT ?
-- bind: 智能手机, 智能手机, 20

{0} 是 jOOQ plain SQL 模板占位符,作为 PreparedStatement 参数传递,无 SQL 注入风险。切勿字符串拼接。

27.6.2 数组聚合

public List<CustomerProducts> getCustomerBoughtProducts() {
    return dsl.select(
        CUSTOMERS.NAME,
        field("array_agg(DISTINCT {0} ORDER BY {0})", String[].class, PRODUCTS.NAME)
            .as("bought_products"))
    .from(CUSTOMERS)
    .join(ORDERS).on(ORDERS.CUSTOMER_ID.eq(CUSTOMERS.ID))
    .join(ORDER_ITEMS).on(ORDER_ITEMS.ORDER_ID.eq(ORDERS.ID))
    .join(PRODUCTS).on(ORDER_ITEMS.PRODUCT_ID.eq(PRODUCTS.ID))
    .where(ORDERS.STATUS.eq("COMPLETED"))
    .groupBy(CUSTOMERS.ID, CUSTOMERS.NAME)
    .orderBy(CUSTOMERS.NAME)
    .fetchInto(CustomerProducts.class);
}

public class JsonbBinding implements Binding<JSONB, Map<String, Object>> {
    private static final ObjectMapper mapper = new ObjectMapper();

    @Override public Converter<JSONB, Map<String, Object>> converter() {
        return new Converter<>() {
            @Override public Map<String, Object> from(JSONB db) {
                if (db == null || db.data() == null) return null;
                try { return mapper.readValue(db.data(), new TypeReference<>() {}); }
                catch (Exception e) { throw new RuntimeException(e); }
            }
            @Override public JSONB to(Map<String, Object> u) {
                if (u == null) return null;
                try { return JSONB.valueOf(mapper.writeValueAsString(u)); }
                catch (Exception e) { throw new RuntimeException(e); }
            }
            @Override public Class<JSONB> fromType() { return JSONB.class; }
            @Override public Class<Map<String, Object>> toType() { return (Class) Map.class; }
        };
    }

    @Override public void sql(BindingSQLContext<Map<String, Object>> ctx) {
        ctx.render().visit(DSL.val(ctx.convert(converter()).value())).sql("::jsonb");
    }
    @Override public void register(BindingRegisterContext<Map<String, Object>> ctx) throws SQLException {
        ctx.statement().registerOutParameter(ctx.index(), Types.VARCHAR);
    }
    @Override public void set(BindingSetStatementContext<Map<String, Object>> ctx) throws SQLException {
        ctx.statement().setString(ctx.index(),
            Objects.toString(ctx.convert(converter()).value(), null));
    }
    @Override public void get(BindingGetResultSetContext<Map<String, Object>> ctx) throws SQLException {
        ctx.convert(converter()).value(JSONB.valueOf(ctx.resultSet().getString(ctx.index())));
    }
}

private static final Map<String, Field<?>> SORT_FIELDS = Map.of(
    "id",       ORDERS.ID,
    "amount",   ORDERS.TOTAL_AMOUNT,
    "created",  ORDERS.CREATED_AT,
    "status",   ORDERS.STATUS,
    "customer", CUSTOMERS.NAME);

private SortField<?> resolveSortField(OrderSearchRequest req) {
    Field<?> sortField = SORT_FIELDS
        .getOrDefault(req.sortField(), ORDERS.CREATED_AT);
    SortOrder order = "asc".equalsIgnoreCase(req.sortOrder())
        ? SortOrder.ASC : SortOrder.DESC;
    return sortField.sort(order);
}

27.8.2 多字段排序

private List<SortField<?>> resolveMultiSort(String sortSpec) {
    if (sortSpec == null || sortSpec.isBlank())
        return List.of(ORDERS.CREATED_AT.desc());

    var sortFields = new ArrayList<SortField<?>>();
    for (String part : sortSpec.split(",")) {
        String[] tokens = part.trim().split(":");
        String name = tokens[0];
        SortOrder order = tokens.length > 1 && "asc".equalsIgnoreCase(tokens[1])
            ? SortOrder.ASC : SortOrder.DESC;
        Field<?> f = SORT_FIELDS.get(name);
        if (f != null) sortFields.add(f.sort(order));
    }
    return sortFields.isEmpty() ? List.of(ORDERS.CREATED_AT.desc()) : sortFields;
}
