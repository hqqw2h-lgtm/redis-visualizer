26.6.1 完整 Repository

package com.example.ecommerce.repository;

import com.example.ecommerce.generated.tables.records.CustomersRecord;
import org.jooq.DSLContext;
import org.jooq.Result;
import org.jooq.Record2;
import org.springframework.stereotype.Repository;

import java.time.LocalDateTime;
import java.util.List;
import java.util.Optional;

import static com.example.ecommerce.generated.Tables.CUSTOMERS;
import static org.jooq.impl.DSL.*;

@Repository
public class CustomerRepository {
    private final DSLContext dsl;

    public CustomerRepository(DSLContext dsl) { this.dsl = dsl; }

    public CustomersRecord create(String name, String email, String phone) {
        return dsl.insertInto(CUSTOMERS)
            .set(CUSTOMERS.NAME, name)
            .set(CUSTOMERS.EMAIL, email)
            .set(CUSTOMERS.PHONE, phone)
            .returning()
            .fetchOne();
    }

    public Optional<CustomersRecord> findById(Long id) {
        return dsl.selectFrom(CUSTOMERS)
            .where(CUSTOMERS.ID.eq(id))
            .fetchOptional();
    }

    public Optional<CustomersRecord> findByEmail(String email) {
        return dsl.selectFrom(CUSTOMERS)
            .where(CUSTOMERS.EMAIL.eq(email))
            .fetchOptional();
    }

    public List<CustomersRecord> findAll(int page, int size) {
        return dsl.selectFrom(CUSTOMERS)
            .orderBy(CUSTOMERS.ID)
            .limit(size)
            .offset(page * size)
            .fetch();
    }

    public List<CustomersRecord> searchByName(String keyword) {
        return dsl.selectFrom(CUSTOMERS)
            .where(CUSTOMERS.NAME.containsIgnoreCase(keyword))
            .orderBy(CUSTOMERS.NAME)
            .fetch();
    }

    public int update(Long id, String name, String email) {
        return dsl.update(CUSTOMERS)
            .set(CUSTOMERS.NAME, name)
            .set(CUSTOMERS.EMAIL, email)
            .set(CUSTOMERS.UPDATED_AT, LocalDateTime.now())
            .where(CUSTOMERS.ID.eq(id))
            .execute();
    }

    public int delete(Long id) {
        return dsl.deleteFrom(CUSTOMERS).where(CUSTOMERS.ID.eq(id)).execute();
    }

    public int count() { return dsl.fetchCount(CUSTOMERS); }

    public boolean existsByEmail(String email) {
        return dsl.fetchExists(dsl.selectFrom(CUSTOMERS).where(CUSTOMERS.EMAIL.eq(email)));
    }

    public Result<Record2<String, Integer>> countByEmailDomain() {
        var domain = substring(CUSTOMERS.EMAIL, position(CUSTOMERS.EMAIL, "@").add(1));
        return dsl.select(domain.as("domain"), count().as("cnt"))
            .from(CUSTOMERS)
            .groupBy(domain)
            .orderBy(count().desc())
            .fetch();
    }
}

26.7.2 createOrder 事务

@Repository
public class OrderRepository {
    private final DSLContext dsl;
    public OrderRepository(DSLContext dsl) { this.dsl = dsl; }

    @Transactional
    public OrdersRecord createOrder(Long customerId, List<OrderItemRequest> items) {
        OrdersRecord order = dsl.insertInto(ORDERS)
            .set(ORDERS.CUSTOMER_ID, customerId)
            .set(ORDERS.STATUS, "PENDING")
            .returning()
            .fetchOne();

        BigDecimal total = BigDecimal.ZERO;
        for (var item : items) {
            var product = dsl.selectFrom(PRODUCTS)
                .where(PRODUCTS.ID.eq(item.productId()))
                .fetchOne();
            if (product == null)
                throw new IllegalArgumentException("商品不存在: " + item.productId());

            BigDecimal subtotal = product.getPrice().multiply(BigDecimal.valueOf(item.quantity()));

            int updated = dsl.update(PRODUCTS)
                .set(PRODUCTS.STOCK, PRODUCTS.STOCK.minus(item.quantity()))
                .where(PRODUCTS.ID.eq(item.productId()))
                .and(PRODUCTS.STOCK.ge(item.quantity()))
                .execute();
            if (updated == 0)
                throw new IllegalStateException("库存不足: " + product.getName());

            dsl.insertInto(ORDER_ITEMS)
                .set(ORDER_ITEMS.ORDER_ID, order.getId())
                .set(ORDER_ITEMS.PRODUCT_ID, item.productId())
                .set(ORDER_ITEMS.QUANTITY, item.quantity())
                .set(ORDER_ITEMS.UNIT_PRICE, product.getPrice())
                .set(ORDER_ITEMS.SUBTOTAL, subtotal)
                .execute();

            total = total.add(subtotal);
        }

        dsl.update(ORDERS).set(ORDERS.TOTAL_AMOUNT, total)
           .where(ORDERS.ID.eq(order.getId())).execute();
        order.setTotalAmount(total);
        return order;
    }

    public Optional<OrderWithItems> findOrderWithItems(Long orderId) {
        return dsl.select(
                ORDERS.ID, ORDERS.STATUS, ORDERS.TOTAL_AMOUNT, ORDERS.CREATED_AT,
                CUSTOMERS.NAME.as("customer_name"),
                multiset(
                    select(ORDER_ITEMS.QUANTITY, ORDER_ITEMS.UNIT_PRICE,
                           ORDER_ITEMS.SUBTOTAL, PRODUCTS.NAME.as("product_name"))
                    .from(ORDER_ITEMS)
                    .join(PRODUCTS).on(ORDER_ITEMS.PRODUCT_ID.eq(PRODUCTS.ID))
                    .where(ORDER_ITEMS.ORDER_ID.eq(ORDERS.ID))
                ).as("items").convertFrom(r -> r.map(Records.mapping(OrderItemDto::new))))
            .from(ORDERS)
            .join(CUSTOMERS).on(ORDERS.CUSTOMER_ID.eq(CUSTOMERS.ID))
            .where(ORDERS.ID.eq(orderId))
            .fetchOptional(Records.mapping(OrderWithItems::new));
    }

    public List<OrderWithItems> findOrders(String status, Long customerId,
            LocalDateTime from, LocalDateTime to, int page, int size) {
        var conditions = noCondition();
        if (status != null)     conditions = conditions.and(ORDERS.STATUS.eq(status));
        if (customerId != null) conditions = conditions.and(ORDERS.CUSTOMER_ID.eq(customerId));
        if (from != null)       conditions = conditions.and(ORDERS.CREATED_AT.ge(from));
        if (to != null)         conditions = conditions.and(ORDERS.CREATED_AT.le(to));

        return dsl.select(
                ORDERS.ID, ORDERS.STATUS, ORDERS.TOTAL_AMOUNT, ORDERS.CREATED_AT,
                CUSTOMERS.NAME.as("customer_name"),
                multiset(
                    select(ORDER_ITEMS.QUANTITY, ORDER_ITEMS.UNIT_PRICE,
                           ORDER_ITEMS.SUBTOTAL, PRODUCTS.NAME.as("product_name"))
                    .from(ORDER_ITEMS)
                    .join(PRODUCTS).on(ORDER_ITEMS.PRODUCT_ID.eq(PRODUCTS.ID))
                    .where(ORDER_ITEMS.ORDER_ID.eq(ORDERS.ID))
                ).as("items").convertFrom(r -> r.map(Records.mapping(OrderItemDto::new))))
            .from(ORDERS)
            .join(CUSTOMERS).on(ORDERS.CUSTOMER_ID.eq(CUSTOMERS.ID))
            .where(conditions)
            .orderBy(ORDERS.CREATED_AT.desc())
            .limit(size).offset(page * size)
            .fetch(Records.mapping(OrderWithItems::new));
    }

    public int updateStatus(Long orderId, String newStatus) {
        return dsl.update(ORDERS)
            .set(ORDERS.STATUS, newStatus)
            .set(ORDERS.UPDATED_AT, LocalDateTime.now())
            .where(ORDERS.ID.eq(orderId))
            .execute();
    }
}

@RestController
@RequestMapping("/api/customers")
public class CustomerController {
    private final CustomerRepository repository;
    public CustomerController(CustomerRepository r) { this.repository = r; }

    @GetMapping
    public List<CustomerResponse> list(@RequestParam(defaultValue = "0") int page,
                                       @RequestParam(defaultValue = "20") int size) {
        return repository.findAll(page, size).stream()
            .map(CustomerResponse::from).toList();
    }

    @GetMapping("/{id}")
    public ResponseEntity<CustomerResponse> getById(@PathVariable Long id) {
        return repository.findById(id)
            .map(CustomerResponse::from)
            .map(ResponseEntity::ok)
            .orElse(ResponseEntity.notFound().build());
    }

    @PostMapping
    public ResponseEntity<CustomerResponse> create(@Valid @RequestBody CustomerRequest req) {
        var record = repository.create(req.name(), req.email(), req.phone());
        return ResponseEntity.status(HttpStatus.CREATED).body(CustomerResponse.from(record));
    }

    @PutMapping("/{id}")
    public ResponseEntity<Void> update(@PathVariable Long id,
                                       @Valid @RequestBody CustomerRequest req) {
        int updated = repository.update(id, req.name(), req.email());
        return updated > 0 ? ResponseEntity.ok().build() : ResponseEntity.notFound().build();
    }

    @DeleteMapping("/{id}")
    public ResponseEntity<Void> delete(@PathVariable Long id) {
        int deleted = repository.delete(id);
        return deleted > 0 ? ResponseEntity.noContent().build() : ResponseEntity.notFound().build();
    }

    @GetMapping("/search")
    public List<CustomerResponse> search(@RequestParam String q) {
        return repository.searchByName(q).stream().map(CustomerResponse::from).toList();
    }
}

26.9 DTO 与 Record 映射

表 26-9:三种映射策略

策略 写法 优点 风险
手动转换 CustomerResponse.from(record) 完全控制、编译时安全 样板多
fetchInto(Class) record.into(CustomerResponse.class) 最简洁 反射、运行时失败
Records.mapping() Records.mapping(Dto::new) 类型安全、MULTISET 友好 字段顺序需匹配

26.9.1 三种策略示例

public record CustomerResponse(Long id, String name, String email) {
    public static CustomerResponse from(CustomersRecord r) {
        return new CustomerResponse(r.getId(), r.getName(), r.getEmail());
    }
}

List<CustomerResponse> a = dsl.selectFrom(CUSTOMERS).fetch().stream()
    .map(CustomerResponse::from).toList();

List<CustomerResponse> b = dsl.selectFrom(CUSTOMERS).fetchInto(CustomerResponse.class);

List<CustomerResponse> c = dsl
    .select(CUSTOMERS.ID, CUSTOMERS.NAME, CUSTOMERS.EMAIL)
    .from(CUSTOMERS)
    .fetch(Records.mapping(CustomerResponse::new));

26.9.2 嵌套 MULTISET 映射

record ItemSummary(Integer quantity, String productName) {}
record OrderSummary(Long id, String status, List<ItemSummary> items) {}

var summaries = dsl.select(
    ORDERS.ID, ORDERS.STATUS,
    multiset(
        select(ORDER_ITEMS.QUANTITY, PRODUCTS.NAME)
        .from(ORDER_ITEMS)
        .join(PRODUCTS).on(ORDER_ITEMS.PRODUCT_ID.eq(PRODUCTS.ID))
        .where(ORDER_ITEMS.ORDER_ID.eq(ORDERS.ID))
    ).as("items").convertFrom(r -> r.map(Records.mapping(ItemSummary::new))))
.from(ORDERS).fetch(Records.mapping(OrderSummary::new));

26.10.1 ErrorResponse + Advice

public record ErrorResponse(int status, String error, String message,
                            String path, LocalDateTime timestamp) {
    public ErrorResponse(int status, String error, String msg, String path) {
        this(status, error, msg, path, LocalDateTime.now());
    }
}

@ControllerAdvice
public class GlobalExceptionHandler {
    @ExceptionHandler(DataAccessException.class)
    public ResponseEntity<ErrorResponse> jooq(DataAccessException ex, HttpServletRequest req) {
        return ResponseEntity.status(HttpStatus.INTERNAL_SERVER_ERROR)
            .body(new ErrorResponse(500, "Database Error", ex.getMessage(), req.getRequestURI()));
    }

    @ExceptionHandler(DuplicateKeyException.class)
    public ResponseEntity<ErrorResponse> duplicate(DuplicateKeyException ex, HttpServletRequest req) {
        return ResponseEntity.status(HttpStatus.CONFLICT)
            .body(new ErrorResponse(409, "Duplicate Entry", "资源已存在", req.getRequestURI()));
    }

    @ExceptionHandler(DataIntegrityViolationException.class)
    public ResponseEntity<ErrorResponse> integrity(DataIntegrityViolationException ex, HttpServletRequest req) {
        return ResponseEntity.badRequest().body(new ErrorResponse(
            400, "Data Integrity Violation",
            "数据约束校验失败: " + extract(ex), req.getRequestURI()));
    }

    @ExceptionHandler(MethodArgumentNotValidException.class)
    public ResponseEntity<ErrorResponse> validation(MethodArgumentNotValidException ex, HttpServletRequest req) {
        String msg = ex.getBindingResult().getFieldErrors().stream()
            .map(f -> f.getField() + ": " + f.getDefaultMessage())
            .reduce((a, b) -> a + "; " + b).orElse("参数校验失败");
        return ResponseEntity.badRequest()
            .body(new ErrorResponse(400, "Validation Failed", msg, req.getRequestURI()));
    }

    @ExceptionHandler(IllegalArgumentException.class)
    public ResponseEntity<ErrorResponse> arg(IllegalArgumentException ex, HttpServletRequest req) {
        return ResponseEntity.badRequest()
            .body(new ErrorResponse(400, "Bad Request", ex.getMessage(), req.getRequestURI()));
    }

    @ExceptionHandler(IllegalStateException.class)
    public ResponseEntity<ErrorResponse> state(IllegalStateException ex, HttpServletRequest req) {
        return ResponseEntity.status(HttpStatus.CONFLICT)
            .body(new ErrorResponse(409, "State Conflict", ex.getMessage(), req.getRequestURI()));
    }

    private String extract(DataIntegrityViolationException ex) {
        String msg = ex.getMostSpecificCause().getMessage();
        if (msg.contains("violates foreign key")) return "外键约束冲突";
        if (msg.contains("violates unique"))      return "唯一约束冲突";
        if (msg.contains("violates not-null"))    return "非空约束冲突";
        return msg;
    }
}

26.13.1 INSERT 调用栈

CustomerController.create()
  -> CustomerService.createCustomer()
     -> CustomerRepository.create()
        -> DefaultDSLContext.insertInto(CUSTOMERS)
        -> InsertImpl.set(...).returning()
        -> InsertImpl.fetchOne()
        -> RenderContext 生成 PostgreSQL SQL
        -> BindContext 绑定 name/email/phone
        -> DataSourceConnectionProvider 获取连接
        -> SpringTransactionProvider 复用事务连接
        -> JDBC PreparedStatement.execute()
        -> ResultSet 映射为 CustomersRecord

设计取舍

取舍 1:Repository 持有 DSLContext

代价是 Repository 与 jOOQ 类型耦合;收益是无需手动创建 Configuration,Boot 自动配置即可使用。

取舍 2:@Transactional 优先于 dsl.transactionResult

代价是依赖 Spring 容器;收益是事务传播无缝、与 Spring Data 共存。

取舍 3:Record 不跨出 Repository

代价是需要 DTO 映射;收益是 Web 层只暴露不可变对象,无副作用。

取舍 4:三种映射策略并存

代价是开发者需选择;收益是覆盖快速原型(fetchInto)、严格安全(手动)、嵌套 MULTISET(Records.mapping)。
