Kotlin 自 Spring Framework 5.0 起被提升为与 Java 并列的一级支持语言。Spring 团队不仅提供扩展函数、DSL 与协程集成,还在框架内部源码中引入 @Nullable 注解适配 Kotlin 类型系统。Kotlin 简洁、安全、与 Java 100% 互操作,但默认 final、data class 缺无参构造、协程与 Reactor 的语义差异,都需要框架层的深度适配。

33.1.3 编译器插件:为什么不可或缺

Kotlin 两个语言特性与 Spring 框架根本性冲突:类默认 final + 主构造函数强制参数。不通过编译器插件解决,Spring 的 AOP 代理与 JPA 实体映射将无法工作。

33.1.5 控制器示例

@RestController
@RequestMapping("/api/orders")
class OrderController(private val orderService: OrderService) {

    @GetMapping("/{id}")
    fun findById(@PathVariable id: Long): ResponseEntity<OrderDto> =
        orderService.findById(id)
            ?.let { ResponseEntity.ok(it.toDto()) }
            ?: ResponseEntity.notFound().build()

    @GetMapping
    fun findAll(
        @RequestParam(defaultValue = "0") page: Int,
        @RequestParam(defaultValue = "20") size: Int
    ): Page<OrderDto> =
        orderService.findAll(PageRequest.of(page, size))
            .map { it.toDto() }

    @PostMapping
    @ResponseStatus(HttpStatus.CREATED)
    fun create(@Valid @RequestBody request: CreateOrderRequest): OrderDto =
        orderService.create(request).toDto()

    @PutMapping("/{id}")
    fun update(
        @PathVariable id: Long,
        @Valid @RequestBody request: UpdateOrderRequest
    ): OrderDto = orderService.update(id, request).toDto()

    @DeleteMapping("/{id}")
    @ResponseStatus(HttpStatus.NO_CONTENT)
    fun delete(@PathVariable id: Long) = orderService.delete(id)
}

data class CreateOrderRequest(
    @field:NotBlank val name: String,
    @field:Email val email: String
)

data class UpdateOrderRequest(
    @field:NotBlank val name: String?,
    @field:Email val email: String?
)

Spring Framework 为 Kotlin 提供大量扩展函数,利用 reified 泛型、lambda、中缀函数等特性,构建类型安全 DSL,提供比注解驱动更灵活的编程模型。

33.2.2 RouterFunctionDsl WebFlux 函数式路由

@Configuration
class RouterConfig {

    @Bean
    fun orderRouter(handler: OrderHandler) = router {
        "/api/orders".nest {
            GET("", handler::findAll)
            GET("/{id}", handler::findById)
            POST("", handler::create)
            PUT("/{id}", handler::update)
            DELETE("/{id}", handler::delete)
        }

        "/api/v1/admin".nest {
            filter { request, next ->
                val token = request.headers().firstHeader("Authorization")
                if (token != null && authService.validate(token)) {
                    next.handle(request)
                } else {
                    ServerResponse.status(HttpStatus.UNAUTHORIZED).buildAndAwait()
                }
            }
            GET("/api/orders", handler::findAll)
            DELETE("/api/orders/{id}", handler::delete)
        }

        accept(MediaType.APPLICATION_JSON).nest {
            GET("/api/v1/search", handler::search)
        }
    }
}

33.3 协程与响应式

Kotlin 协程为异步编程提供优雅抽象:用同步风格代码实现非阻塞行为。Spring Framework 5.2 集成 Kotlin 协程,suspend 函数替代 Mono Flux 操作符链。理解底层转换机制才能避免性能陷阱与并发 bug。

表 33-3-1:协程与 Reactor 类型映射

Kotlin 协程写法 Reactor 语义 适合返回
suspend fun foo(): T Mono<T> 单个非空值
suspend fun foo(): T? Mono<T> 或 empty 可能不存在的单值
fun foo(): Flow<T> Flux<T> 多个值 / 流式响应
suspend fun foo(): Unit Mono<Void> 只关心完成信号

33.3.1 suspend 函数在 @RestController

@RestController
@RequestMapping("/api/orders")
class OrderCoroutineController(
    private val orderService: OrderService,
    private val notificationService: NotificationService
) {

    @GetMapping("/{id}")
    suspend fun findById(@PathVariable id: Long): Order =
        orderService.findById(id)
            ?: throw ResponseStatusException(HttpStatus.NOT_FOUND)

    @GetMapping
    fun findAll(): Flow<Order> = orderService.findAll()

    @PostMapping
    suspend fun create(@RequestBody request: CreateOrderRequest): ResponseEntity<Order> {
        val order = orderService.create(request)
        return ResponseEntity
            .created(URI.create("/api/orders/${order.id}"))
            .body(order)
    }

    @GetMapping("/{id}/dashboard")
    suspend fun dashboard(@PathVariable id: Long): DashboardDto =
        coroutineScope {
            val orderDeferred = async { orderService.findById(id) }
            val historyDeferred = async { orderService.findByCustomerId(id) }
            val notifDeferred = async { notificationService.countUnread(id) }

            val order = orderDeferred.await()
                ?: throw ResponseStatusException(HttpStatus.NOT_FOUND)

            DashboardDto(
                order = order.toDto(),
                recentOrders = historyDeferred.await().map { it.toDto() },
                unreadNotifications = notifDeferred.await()
            )
        }
}

data class DashboardDto(
    val order: OrderDto,
    val recentOrders: List<OrderDto>,
    val unreadNotifications: Int
)

suspend 只改变服务端执行模型,不改变客户端契约。

33.3.2 CoRouter 协程风格函数式路由

@Configuration
class CoRouterConfig {

    @Bean
    fun orderCoRouter(handler: OrderCoroutineHandler) = coRouter {
        "/api/orders".nest {
            GET("", handler::findAll)
            GET("/{id}", handler::findById)
            POST("", handler::create)
            PUT("/{id}", handler::update)
            DELETE("/{id}", handler::delete)
        }

        filter { request, next ->
            val start = System.currentTimeMillis()
            val response = next(request)
            val elapsed = System.currentTimeMillis() - start
            logger.info("${request.method()} ${request.path()} -> ${elapsed}ms")
            response
        }
    }
}

@Component
class OrderCoroutineHandler(
    private val orderService: OrderService
) {
    suspend fun findAll(request: ServerRequest): ServerResponse {
        val orders = orderService.findAll()
        return ServerResponse.ok()
            .contentType(MediaType.APPLICATION_JSON)
            .bodyAndAwait(orders)
    }

    suspend fun findById(request: ServerRequest): ServerResponse {
        val id = request.pathVariable("id").toLong()
        return orderService.findById(id)
            ?.let { ServerResponse.ok().bodyValueAndAwait(it.toDto()) }
            ?: ServerResponse.notFound().buildAndAwait()
    }

    suspend fun create(request: ServerRequest): ServerResponse {
        val body = request.awaitBody<CreateOrderRequest>()
        val order = orderService.create(body)
        return ServerResponse
            .created(URI.create("/api/orders/${order.id}"))
            .bodyValueAndAwait(order.toDto())
    }
}

33.3.3 R2DBC 与 CoroutineCrudRepository

interface OrderRepository : CoroutineCrudRepository<Order, Long> {

    suspend fun findByEmail(email: String): Order?

    fun findByStatus(status: OrderStatus): Flow<Order>

    @Query("SELECT * FROM orders WHERE name LIKE :pattern")
    fun searchByName(pattern: String): Flow<Order>

    suspend fun findByStatus(status: OrderStatus, pageable: Pageable): Flow<Order>
}

@Service
class OrderService(private val orderRepo: OrderRepository) {

    suspend fun findById(id: Long): Order? = orderRepo.findById(id)

    fun findAll(): Flow<Order> = orderRepo.findAll()

    suspend fun create(request: CreateOrderRequest): Order {
        val order = Order(
            name = request.name,
            email = request.email
        )
        return orderRepo.save(order)
    }

    suspend fun update(id: Long, request: UpdateOrderRequest): Order {
        val order = orderRepo.findById(id)
            ?: throw ResponseStatusException(HttpStatus.NOT_FOUND)

        request.name?.let { order.name = it }
        request.email?.let { order.email = it }

        return orderRepo.save(order)
    }

    suspend fun delete(id: Long) = orderRepo.deleteById(id)

    @Transactional
    suspend fun transferOrder(fromCustomerId: Long, toCustomerId: Long, orderId: Long) {
        val order = orderRepo.findById(orderId)
            ?: throw IllegalArgumentException("Order not found")
        order.customerId = toCustomerId
        orderRepo.save(order)
    }
}

33.4 null 安全与类型系统

Kotlin null 安全是其最重要语言特性:编译期捕获潜在 NPE。但 Kotlin 调用 Java Spring API 时,Java 类型默认是"平台类型"(既非 T 也非 T?)。Spring Framework 通过整个 API 添加 null 安全注解,让 Kotlin 编译器正确推断每个参数和返回值的可空性。

图 33-4-1 · Java @Nullable → Kotlin T? 推断链(JSR-305 strict + 编译期 null 检查)

表 33-4-1:Spring null 安全注解

注解 作用
@Nullable 参数 / 返回值可能为 null
@NonNull 参数 / 返回值不可能为 null
@NonNullApi 包级默认非空(参数和返回值)
@NonNullFields 包级默认非空(字段)

33.4.1 -Xjsr305=strict 配置

tasks.withType<KotlinCompile> {
    kotlinOptions {
        freeCompilerArgs += "-Xjsr305=strict"
    }
}

可选值:ignore(默认 忽略所有 null 注解)、warn(不匹配时编译警告)、strict(不匹配时编译错误,推荐)。

@Service
class OrderService(private val orderRepo: OrderRepository) {

    fun findById(id: Long): Order? {
        return orderRepo.findByIdOrNull(id)
    }

    fun getById(id: Long): Order {
        return orderRepo.findByIdOrNull(id)
            ?: throw OrderNotFoundException("Order $id not found")
    }

    fun create(name: String, email: String): Order {
        val order = Order(name = name, email = email)
        return orderRepo.save(order)
    }
}

33.5.1 MockK Kotlin 原生 Mock 框架

Mockito 在 Kotlin 上的兼容性问题:Kotlin 类默认 final(mockito-inline 才能 mock)、不支持协程 mock、不支持 Kotlin 命名参数 / 默认参数。MockK 专为 Kotlin 设计,原生支持这些特性。

class OrderServiceTest {

    private val orderRepo = mockk<OrderRepository>()
    private val eventPublisher = mockk<ApplicationEventPublisher>(relaxed = true)
    private val orderService = OrderService(orderRepo, eventPublisher)

    @Test
    fun `should find order by id`() {
        val expectedOrder = Order(id = 1, name = "Customer C1001", email = "customer-c1001@example.com")
        every { orderRepo.findByIdOrNull(1) } returns expectedOrder

        val result = orderService.findById(1)

        assertThat(result).isEqualTo(expectedOrder)
        verify(exactly = 1) { orderRepo.findByIdOrNull(1) }
    }

    @Test
    fun `should throw exception when order not found`() {
        every { orderRepo.findByIdOrNull(any()) } returns null

        assertThrows<OrderNotFoundException> {
            orderService.getById(999)
        }
    }

    @Test
    fun `should save order and publish event`() {
        val request = CreateOrderRequest(name = "Order O1001", email = "order-o1001@example.com")
        val savedOrder = Order(id = 1, name = "Order O1001", email = "order-o1001@example.com")

        every { orderRepo.save(any<Order>()) } returns savedOrder
        every { orderRepo.findByEmail("order-o1001@example.com") } returns null

        val result = orderService.create(request)

        assertThat(result.name).isEqualTo("Order O1001")

        verify {
            eventPublisher.publishEvent(
                withArg<OrderCreatedEvent> {
                    assertThat(it.orderId).isEqualTo(1)
                }
            )
        }

        verifyOrder {
            orderRepo.findByEmail("order-o1001@example.com")
            orderRepo.save(any())
            eventPublisher.publishEvent(any<OrderCreatedEvent>())
        }
    }

    @Test
    fun `should mock suspend functions`() = runTest {
        val orderRepo = mockk<CoroutineOrderRepository>()
        coEvery { orderRepo.findById(1) } returns Order(1, "Customer C1001", "customer-c1001@example.com")

        val result = orderRepo.findById(1)
        assertThat(result?.name).isEqualTo("Customer C1001")

        coVerify { orderRepo.findById(1) }
    }
}

33.6 陷阱与限制

33.6.1 陷阱 1 缺 kotlin-spring 插件导致 CGLIB 代理失败

@Service
class OrderService(private val orderRepo: OrderRepository) {

    @Transactional
    fun createOrder(request: CreateOrderRequest): Order {
        val order = orderRepo.save(Order(request))
        if (order.totalAmount > 10000) {
            throw BusinessException("Amount exceeds limit")
        }
        return order
    }
}

Kotlin 类默认 final,@Transactional 通常需 CGLIB 子类代理拦截方法调用。final 类无法被继承,代理创建失败或事务不生效。修复:build.gradle.kts 加 kotlin("plugin.spring") version "1.9.24",Spring 注解类自动 open。验证:javap -p OrderService.class 应看到 public class OrderService(无 final)。

33.6.2 陷阱 2 data class 当 JPA 实体

@Entity
data class Order(
    @Id @GeneratedValue(strategy = GenerationType.IDENTITY)
    val id: Long = 0,
    val name: String,
    val email: String
)

val order = Order(name = "Customer C1001", email = "customer-c1001@example.com")
val set = mutableSetOf(order)
println(order in set)

entityManager.persist(order)
println(order in set)

val managed = entityManager.find(Order::class.java, 1L)
val modified = managed.copy(name = "Order O1001")

@Entity
@Table(name = "orders")
class Order(
    @Id @GeneratedValue(strategy = GenerationType.IDENTITY)
    val id: Long = 0,

    @Column(nullable = false)
    var name: String,

    @Column(nullable = false, unique = true)
    var email: String,

    @Enumerated(EnumType.STRING)
    var status: OrderStatus = OrderStatus.ACTIVE
) {
    override fun equals(other: Any?): Boolean {
        if (this === other) return true
        if (other !is Order) return false
        return email == other.email
    }

    override fun hashCode(): Int = email.hashCode()

    override fun toString(): String = "Order(id=$id, name=$name, email=$email)"
}

data class 自动生成 equals / hashCode 含所有构造参数(包括 id)。新建实体 id=0,保存后 id=1,hashCode 不同,导致它在 HashSet 中"消失"。copy() 创建新对象,绕过 Hibernate 脏检查。修复:Entity 用普通 class,equals / hashCode 基于业务键(如 email)。

33.6.3 陷阱 3 缺 Jackson Kotlin Module 导致反序列化失败

data class CreateOrderRequest(
    val name: String,
    val email: String
)

Jackson 默认通过无参构造 + setter 反序列化 Java 对象。Kotlin data class 只有带参主构造,无无参构造和 setter。修复:implementation("com.fasterxml.jackson.module:jackson-module-kotlin")。Spring Boot 自动配置在 classpath 含 jackson-module-kotlin 时自动注册 KotlinModule。

val objectMapper = jacksonObjectMapper()
val objectMapper2 = ObjectMapper().registerModule(KotlinModule.Builder().build())

KotlinModule 提供:通过 Kotlin 反射读取构造参数名(无需 -parameters 编译选项)、支持 data class 默认参数值、正确处理 Kotlin 可空类型、支持 Kotlin value class。

33.6.4 陷阱 4 协程异常处理与 Mono 异常传播差异

fun findOrderReactive(id: Long): Mono<OrderDto> =
    orderRepo.findById(id)
        .map { it.toDto() }
        .switchIfEmpty(Mono.error(OrderNotFoundException(id)))
        .onErrorResume(DatabaseException::class.java) {
            Mono.just(OrderDto.fallback(id))
        }

suspend fun findOrderCoroutine(id: Long): OrderDto {
    return try {
        val order = orderRepo.findById(id)
            ?: throw OrderNotFoundException(id)
        order.toDto()
    } catch (e: DatabaseException) {
        OrderDto.fallback(id)
    }
}

suspend fun dashboard(customerId: Long): DashboardDto = coroutineScope {
    val order = async { orderService.findById(customerId) }
    val orders = async { orderService.findByCustomer(customerId) }
    val stats = async { statsService.getOrderStats(customerId) }

    DashboardDto(order.await()!!, orders.await(), stats.await())
}

suspend fun dashboardSafe(customerId: Long): DashboardDto = supervisorScope {
    val order = async { orderService.findById(customerId) }
    val orders = async {
        try { orderService.findByCustomer(customerId) }
        catch (e: Exception) { emptyList() }
    }
    val stats = async {
        try { statsService.getOrderStats(customerId) }
        catch (e: Exception) { OrderStats.empty() }
    }

    DashboardDto(
        order = order.await() ?: throw OrderNotFoundException(customerId),
        orders = orders.await(),
        stats = stats.await()
    )
}

Reactor 把异常视为流中"错误信号",通过 onErrorResume onErrorReturn 处理。协程用标准 try-catch。coroutineScope 中任一 async 失败会取消整个 scope;supervisorScope 隔离失败不影响其他子任务。
