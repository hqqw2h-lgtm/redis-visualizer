package com.example.ecommerce.entity;

import jakarta.persistence.*;
import java.time.LocalDateTime;
import java.util.ArrayList;
import java.util.List;

@Entity
@Table(name = "customers")
public class Customer {

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @Column(nullable = false, length = 100)
    private String name;

    @Column(nullable = false, unique = true, length = 200)
    private String email;

    @Column(length = 20)
    private String phone;

    @Column(name = "created_at", nullable = false, updatable = false)
    private LocalDateTime createdAt;

    @Column(name = "updated_at")
    private LocalDateTime updatedAt;

    @OneToMany(mappedBy = "customer")
    private List<Order> orders = new ArrayList<>();

    @PrePersist
    protected void onCreate() {
        this.createdAt = LocalDateTime.now();
        this.updatedAt = this.createdAt;
    }

    @PreUpdate
    protected void onUpdate() {
        this.updatedAt = LocalDateTime.now();
    }

}

Order 实体

package com.example.ecommerce.entity;

import jakarta.persistence.*;
import java.math.BigDecimal;
import java.time.LocalDateTime;
import java.util.ArrayList;
import java.util.List;

@Entity
@Table(name = "orders", indexes = {
    @Index(name = "idx_order_customer", columnList = "customer_id"),
    @Index(name = "idx_order_status", columnList = "status"),
    @Index(name = "idx_order_tenant", columnList = "tenant_id"),
    @Index(name = "idx_order_created", columnList = "created_at DESC")
})
public class Order {

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "customer_id", nullable = false)
    private Customer customer;

    @Enumerated(EnumType.STRING)
    @Column(nullable = false, length = 20)
    private OrderStatus status;

    @Column(name = "total_amount", nullable = false, precision = 12, scale = 2)
    private BigDecimal totalAmount;

    @Column(name = "tenant_id", nullable = false, length = 50)
    private String tenantId;

    @OneToMany(mappedBy = "order", cascade = CascadeType.ALL, orphanRemoval = true)
    private List<OrderItem> items = new ArrayList<>();

    @Column(name = "created_at", nullable = false, updatable = false)
    private LocalDateTime createdAt;

    @Version
    private Integer version;

    @PrePersist
    protected void onCreate() {
        this.createdAt = LocalDateTime.now();
        if (this.status == null) {
            this.status = OrderStatus.PENDING;
        }
    }

    public void addItem(OrderItem item) {
        items.add(item);
        item.setOrder(this);
    }

    public void removeItem(OrderItem item) {
        items.remove(item);
        item.setOrder(null);
    }

    public void recalculateTotal() {
        this.totalAmount = items.stream()
            .map(OrderItem::getSubtotal)
            .reduce(BigDecimal.ZERO, BigDecimal::add);
    }

    // getters & setters ...
}

BlazeConfig.java 完整代码

package com.example.ecommerce.config;

import com.blazebit.persistence.Criteria;
import com.blazebit.persistence.CriteriaBuilderFactory;
import com.blazebit.persistence.spi.CriteriaBuilderConfiguration;
import com.blazebit.persistence.view.EntityViewManager;
import com.blazebit.persistence.view.spi.EntityViewConfiguration;
import com.blazebit.persistence.spring.data.repository.config.EnableBlazeRepositories;
import com.blazebit.persistence.view.EntityViews;
import com.blazebit.persistence.spring.data.impl.repository.BlazePersistenceRepositoryFactoryBean;
import jakarta.persistence.EntityManagerFactory;
import org.springframework.beans.factory.config.ConfigurableBeanFactory;
import org.springframework.context.annotation.*;

@Configuration
@EnableEntityViews(basePackages = "com.example.ecommerce.view")
@EnableBlazeRepositories(
    basePackages = "com.example.ecommerce.repository",
    repositoryFactoryBeanClass = BlazePersistenceRepositoryFactoryBean.class
)
public class BlazeConfig {

    @Bean
    @Scope(ConfigurableBeanFactory.SCOPE_SINGLETON)
    public CriteriaBuilderFactory criteriaBuilderFactory(
            EntityManagerFactory emf) {
        CriteriaBuilderConfiguration config = Criteria.getDefault();
        return config.createCriteriaBuilderFactory(emf);
    }

    @Bean
    @Scope(ConfigurableBeanFactory.SCOPE_SINGLETON)
    public EntityViewManager entityViewManager(
            CriteriaBuilderFactory cbf,
            EntityViewConfiguration entityViewConfiguration) {
        return entityViewConfiguration.createEntityViewManager(cbf);
    }
}
@EnableEntityViews 扫描指定包下所有带 @EntityView 注解的接口/抽象类，自动注册到 EntityViewConfiguration。@EnableBlazeRepositories 将 Spring Data 的 Repository 工厂替换为 Blaze 的实现——所有继承 EntityViewRepository 的接口都会自动注入 EntityViewManager。

22.5 Entity View 定义 — DTO 投影

CustomerService 完整实现

package com.example.ecommerce.service;

import com.blazebit.persistence.view.EntityViewManager;
import com.blazebit.persistence.spring.data.repository.KeysetAwarePage;
import com.blazebit.persistence.spring.data.repository.KeysetPageable;
import com.example.ecommerce.repository.CustomerRepository;
import com.example.ecommerce.view.customer.*;
import jakarta.persistence.EntityManager;
import jakarta.persistence.EntityNotFoundException;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import java.util.List;

@Service
@Transactional
public class CustomerService {

    private final CustomerRepository customerRepo;
    private final EntityViewManager evm;
    private final EntityManager em;

    public CustomerService(CustomerRepository customerRepo,
                           EntityViewManager evm,
                           EntityManager em) {
        this.customerRepo = customerRepo;
        this.evm = evm;
        this.em = em;
    }

    // ---- Create ----
    public CustomerDetailView create(CustomerCreateView createView) {
        evm.save(em, createView);
        em.flush();
        return evm.find(em, CustomerDetailView.class, createView.getId());
    }

    // ---- Read (单条) ----
    @Transactional(readOnly = true)
    public CustomerDetailView findById(Long id) {
        CustomerDetailView view = evm.find(em, CustomerDetailView.class, id);
        if (view == null) {
            throw new EntityNotFoundException(
                "Customer not found: id=" + id);
        }
        return view;
    }

    // ---- Read (列表 + Keyset 分页) ----
    @Transactional(readOnly = true)
    public KeysetAwarePage<CustomerListView> findAll(
            KeysetPageable pageable) {
        return customerRepo.findAll(pageable);
    }

    // ---- Read (搜索) ----
    @Transactional(readOnly = true)
    public List<CustomerListView> searchByName(String keyword) {
        return customerRepo.findByNameContainingIgnoreCase(keyword);
    }

    // ---- Update ----
    public CustomerDetailView update(Long id,
                                     CustomerUpdateView updateView) {
        if (!id.equals(updateView.getId())) {
            throw new IllegalArgumentException(
                "Path id and body id mismatch");
        }
        evm.save(em, updateView);
        em.flush();
        return evm.find(em, CustomerDetailView.class, id);
    }

    // ---- Delete ----
    public void delete(Long id) {
        CustomerDetailView existing = evm.find(
            em, CustomerDetailView.class, id);
        if (existing == null) {
            throw new EntityNotFoundException(
                "Customer not found: id=" + id);
        }
        evm.remove(em, CustomerListView.class, id);
    }
}

OrderService 完整实现

package com.example.ecommerce.service;

import com.blazebit.persistence.CriteriaBuilderFactory;
import com.blazebit.persistence.view.EntityViewManager;
import com.example.ecommerce.dto.CreateOrderRequest;
import com.example.ecommerce.entity.*;
import com.example.ecommerce.repository.OrderRepository;
import com.example.ecommerce.view.order.*;
import jakarta.persistence.EntityManager;
import jakarta.persistence.EntityNotFoundException;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import java.math.BigDecimal;
import java.util.List;
import java.util.Map;

@Service
@Transactional
public class OrderService {

    private final OrderRepository orderRepo;
    private final EntityViewManager evm;
    private final EntityManager em;
    private final CriteriaBuilderFactory cbf;

    public OrderService(OrderRepository orderRepo,
                        EntityViewManager evm,
                        EntityManager em,
                        CriteriaBuilderFactory cbf) {
        this.orderRepo = orderRepo;
        this.evm = evm;
        this.em = em;
        this.cbf = cbf;
    }

    public OrderDetailView createOrder(CreateOrderRequest req) {
        Order order = new Order();
        order.setCustomer(em.getReference(Customer.class,
                                          req.getCustomerId()));
        order.setTenantId(req.getTenantId());
        order.setStatus(OrderStatus.PENDING);

        for (CreateOrderRequest.ItemRequest item : req.getItems()) {
            Product product = em.find(Product.class,
                                      item.getProductId());
            if (product == null) {
                throw new EntityNotFoundException(
                    "Product not found: " + item.getProductId());
            }
            if (product.getStockQuantity() < item.getQuantity()) {
                throw new IllegalStateException(
                    "Insufficient stock for: " + product.getSku());
            }

            OrderItem orderItem = new OrderItem();
            orderItem.setProduct(product);
            orderItem.setQuantity(item.getQuantity());
            orderItem.setUnitPrice(product.getPrice());
            orderItem.setSubtotal(product.getPrice()
                .multiply(BigDecimal.valueOf(item.getQuantity())));
            order.addItem(orderItem);

            product.setStockQuantity(
                product.getStockQuantity() - item.getQuantity());
        }

        order.recalculateTotal();
        em.persist(order);
        em.flush();

        return evm.find(em, OrderDetailView.class, order.getId());
    }

    @Transactional(readOnly = true)
    public OrderDetailView findById(Long id) {
        OrderDetailView view = evm.find(
            em, OrderDetailView.class, id);
        if (view == null) {
            throw new EntityNotFoundException(
                "Order not found: id=" + id);
        }
        return view;
    }

    @Transactional(readOnly = true)
    public List<OrderSummaryView> findByCustomer(Long customerId) {
        return orderRepo.findByCustomerId(customerId);
    }

    // 订单状态机推进
    private static final Map<OrderStatus, List<OrderStatus>>
        VALID_TRANSITIONS = Map.of(
            OrderStatus.PENDING,    List.of(OrderStatus.PAID,
                                            OrderStatus.CANCELLED),
            OrderStatus.PAID,       List.of(OrderStatus.PROCESSING,
                                            OrderStatus.REFUNDED),
            OrderStatus.PROCESSING, List.of(OrderStatus.SHIPPED),
            OrderStatus.SHIPPED,    List.of(OrderStatus.DELIVERED),
            OrderStatus.DELIVERED,  List.of(OrderStatus.REFUNDED)
        );

    public OrderDetailView updateStatus(Long id,
                                        OrderStatus newStatus) {
        Order order = em.find(Order.class, id);
        if (order == null) {
            throw new EntityNotFoundException(
                "Order not found: id=" + id);
        }

        List<OrderStatus> allowed = VALID_TRANSITIONS
            .getOrDefault(order.getStatus(), List.of());
        if (!allowed.contains(newStatus)) {
            throw new IllegalStateException(String.format(
                "Invalid transition: %s → %s. Allowed: %s",
                order.getStatus(), newStatus, allowed));
        }

        order.setStatus(newStatus);
        em.flush();

        return evm.find(em, OrderDetailView.class, id);
    }
}

package com.example.ecommerce.controller;

import com.blazebit.persistence.spring.data.repository.KeysetAwarePage;
import com.blazebit.persistence.spring.data.repository.KeysetPageRequest;
import com.blazebit.persistence.spring.data.repository.KeysetPageable;
import com.example.ecommerce.dto.CustomerCreateRequest;
import com.example.ecommerce.service.CustomerService;
import com.example.ecommerce.view.customer.*;
import jakarta.validation.Valid;
import org.springframework.data.domain.Sort;
import org.springframework.http.HttpStatus;
import org.springframework.web.bind.annotation.*;
import java.util.List;

@RestController
@RequestMapping("/api/customers")
public class CustomerController {

    private final CustomerService customerService;

    public CustomerController(CustomerService customerService) {
        this.customerService = customerService;
    }

    @GetMapping
    public KeysetAwarePage<CustomerListView> list(
            @RequestParam(defaultValue = "0") int page,
            @RequestParam(defaultValue = "20") int size) {
        KeysetPageable pageable = KeysetPageRequest.of(
            page, size, Sort.by(Sort.Direction.DESC, "id"));
        return customerService.findAll(pageable);
    }

    @GetMapping("/{id}")
    public CustomerDetailView getById(@PathVariable Long id) {
        return customerService.findById(id);
    }

    @GetMapping("/search")
    public List<CustomerListView> search(
            @RequestParam String name) {
        return customerService.searchByName(name);
    }

    @PostMapping
    @ResponseStatus(HttpStatus.CREATED)
    public CustomerDetailView create(
            @Valid @RequestBody CustomerCreateRequest req) {
        CustomerCreateView createView = customerService
            .getEntityViewManager()
            .create(CustomerCreateView.class);
        createView.setName(req.getName());
        createView.setEmail(req.getEmail());
        createView.setPhone(req.getPhone());
        return customerService.create(createView);
    }

    @PutMapping("/{id}")
    public CustomerDetailView update(
            @PathVariable Long id,
            @Valid @RequestBody CustomerCreateRequest req) {
        CustomerUpdateView updateView = customerService
            .getEntityViewManager()
            .find(customerService.getEntityManager(),
                  CustomerUpdateView.class, id);
        if (updateView == null) {
            throw new jakarta.persistence.EntityNotFoundException(
                "Customer not found: id=" + id);
        }
        updateView.setName(req.getName());
        updateView.setEmail(req.getEmail());
        return customerService.update(id, updateView);
    }

    @DeleteMapping("/{id}")
    @ResponseStatus(HttpStatus.NO_CONTENT)
    public void delete(@PathVariable Long id) {
        customerService.delete(id);
    }
}

GlobalExceptionHandler

package com.example.ecommerce.exception;

import com.example.ecommerce.dto.ErrorResponse;
import jakarta.persistence.EntityNotFoundException;
import jakarta.persistence.OptimisticLockException;
import jakarta.servlet.http.HttpServletRequest;
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.MethodArgumentNotValidException;
import org.springframework.web.bind.annotation.*;
import java.util.stream.Collectors;

@RestControllerAdvice
public class GlobalExceptionHandler {

    @ExceptionHandler(EntityNotFoundException.class)
    public ResponseEntity<ErrorResponse> handleNotFound(
            EntityNotFoundException ex,
            HttpServletRequest request) {
        return ResponseEntity.status(HttpStatus.NOT_FOUND)
            .body(new ErrorResponse(
                404, "Not Found",
                ex.getMessage(),
                request.getRequestURI()));
    }

    @ExceptionHandler(OptimisticLockException.class)
    public ResponseEntity<ErrorResponse> handleConflict(
            OptimisticLockException ex,
            HttpServletRequest request) {
        return ResponseEntity.status(HttpStatus.CONFLICT)
            .body(new ErrorResponse(
                409, "Conflict",
                "Resource was modified by another request. "
                + "Please retry with the latest version.",
                request.getRequestURI()));
    }

    @ExceptionHandler(IllegalStateException.class)
    public ResponseEntity<ErrorResponse> handleBadState(
            IllegalStateException ex,
            HttpServletRequest request) {
        return ResponseEntity.status(HttpStatus.UNPROCESSABLE_ENTITY)
            .body(new ErrorResponse(
                422, "Unprocessable Entity",
                ex.getMessage(),
                request.getRequestURI()));
    }

    @ExceptionHandler(IllegalArgumentException.class)
    public ResponseEntity<ErrorResponse> handleBadArgument(
            IllegalArgumentException ex,
            HttpServletRequest request) {
        return ResponseEntity.status(HttpStatus.BAD_REQUEST)
            .body(new ErrorResponse(
                400, "Bad Request",
                ex.getMessage(),
                request.getRequestURI()));
    }

    @ExceptionHandler(MethodArgumentNotValidException.class)
    public ResponseEntity<ErrorResponse> handleValidation(
            MethodArgumentNotValidException ex,
            HttpServletRequest request) {
        String errors = ex.getBindingResult()
            .getFieldErrors().stream()
            .map(e -> e.getField() + ": " + e.getDefaultMessage())
            .collect(Collectors.joining("; "));
        return ResponseEntity.status(HttpStatus.BAD_REQUEST)
            .body(new ErrorResponse(
                400, "Validation Failed",
                errors,
                request.getRequestURI()));
    }

    @ExceptionHandler(Exception.class)
    public ResponseEntity<ErrorResponse> handleGeneric(
            Exception ex,
            HttpServletRequest request) {
        return ResponseEntity
            .status(HttpStatus.INTERNAL_SERVER_ERROR)
            .body(new ErrorResponse(
                500, "Internal Server Error",
                "An unexpected error occurred",
                request.getRequestURI()));
    }
}

22.10 Entity View vs 原生 JPA 对比

最直观的对比方式是"做同样的事，看代码量和 SQL 质量的差异"。以下 4 个场景展示了 Entity View 的优势。

场景 1：列表查询

// 纯 JPA：Repository
public interface CustomerJpaRepo
    extends JpaRepository<Customer, Long> {
}

// 纯 JPA：DTO
public record CustomerDTO(
    Long id, String name, String email,
    long orderCount) {}

// 纯 JPA：Service
public List<CustomerDTO> listCustomers() {
    return customerJpaRepo.findAll().stream()
        .map(c -> new CustomerDTO(
            c.getId(),
            c.getName(),
            c.getEmail(),
            c.getOrders().size()  // N+1 !
        ))
        .toList();
}

// 生成 SQL：
// 1. SELECT * FROM customers        (全列)
// 2. SELECT * FROM orders WHERE customer_id=1
// 3. SELECT * FROM orders WHERE customer_id=2
// 4. SELECT * FROM orders WHERE customer_id=3
// ... N+1 条 SQL !
// Blaze：View 声明
@EntityView(Customer.class)
public interface CustomerListView {
    @IdMapping Long getId();
    String getName();
    String getEmail();
    @Mapping("SIZE(orders)")
    long getOrderCount();
}

// Blaze：Repository
public interface CustomerRepository
    extends EntityViewRepository
            <CustomerListView, Long> {}

// Blaze：Service
public List<CustomerListView> list() {
    return customerRepo.findAll();
}

// 生成 SQL（仅 1 条）：
// SELECT c.id, c.name, c.email,
//   (SELECT COUNT(*)
//    FROM orders o
//    WHERE o.customer_id = c.id)
// FROM customers c

场景 2：创建操作

// 纯 JPA：
public Customer create(CustomerDTO dto) {
    Customer c = new Customer();
    c.setName(dto.name());
    c.setEmail(dto.email());
    c.setPhone(dto.phone());
    Customer saved = customerJpaRepo.save(c);
    // 需要手动转回 DTO
    return saved;
}
// 返回的是 Entity，序列化会暴露所有字段
// 或需要再转一次 DTO
// Blaze：
public CustomerDetailView create(
        CustomerCreateView view) {
    evm.save(em, view);
    em.flush();
    return evm.find(em,
        CustomerDetailView.class,
        view.getId());
}
// 返回 View，序列化精确控制
// CreateView 只暴露可写字段
// DetailView 只暴露可读字段

场景 3：更新操作

// 纯 JPA：read-modify-write
public Customer update(Long id, CustomerDTO dto) {
    Customer c = customerJpaRepo.findById(id)
        .orElseThrow(() -> new EntityNotFoundException(
            "Customer not found"));
    c.setName(dto.name());    // 手动逐字段赋值
    c.setEmail(dto.email());
    return customerJpaRepo.save(c);
}
// SQL: SELECT * FROM customers WHERE id=?
//      UPDATE customers SET name=?, email=?,
//        phone=?, created_at=?, updated_at=?
//      WHERE id=?
// ↑ 即使 phone 未改，也 SET 了全部列
// Blaze：脏追踪 + 精确更新
public CustomerDetailView update(Long id,
        CustomerUpdateView view) {
    evm.save(em, view);
    em.flush();
    return evm.find(em,
        CustomerDetailView.class, id);
}
// SQL: UPDATE customers
//      SET name=?, email=?
//      WHERE id=?
// ↑ 只更新 UpdateView 声明的字段
// 脏追踪只 SET 实际变更的列

场景 4：嵌套关联查询

// 纯 JPA：EntityGraph
@EntityGraph(attributePaths = {
    "customer", "items", "items.product"})
Optional<Order> findById(Long id);

// 返回 Order Entity——包含全部字段
// 如果 items 有 100 条，
// items.product 也全量加载
// 序列化需手动裁剪或用 @JsonIgnore
// Blaze：嵌套 View 精确投影

// JOIN customers c ON ...
// LEFT JOIN order_items i ON ...
// LEFT JOIN products p ON ...

22.11 陷阱警示

即使 Blaze-Persistence 消除了大量样板代码，在实际使用中仍有几个容易踩坑的地方。本节列举 4 类常见陷阱及其解决方案。

陷阱 1：LazyInitializationException 与 Entity View

问题：在 @Transactional(readOnly = true) 的 Service 方法中返回 Entity View，View 中包含 @Mapping("orders") 关联——Controller 层序列化 JSON 时，事务已关闭，触发 LazyInitializationException。 
// ❌ 错误：View 在 Controller 层序列化时事务已关闭
@EntityView(Customer.class)
public interface CustomerBadView {
    @IdMapping Long getId();
    List<Order> getOrders();  // 返回 Entity 而非 View！
}

// ✅ 正确：关联字段必须返回另一个 Entity View
@EntityView(Customer.class)
public interface CustomerGoodView {
    @IdMapping Long getId();
    List<OrderSummaryView> getOrders();  // View → View
}

// ✅ 或者用 @Mapping 做内联子查询
@EntityView(Customer.class)
public interface CustomerCountView {
    @IdMapping Long getId();
    @Mapping("SIZE(orders)")
    long getOrderCount();  // 在 SQL 层面计算，不触发延迟加载
}

陷阱 2：UpdatableEntityView 的 flush 顺序

问题：同时修改多个 UpdatableEntityView 时，evm.save() 的调用顺序会影响 SQL 执行顺序——如果存在外键依赖，顺序不当会导致约束违反。 
// ❌ 错误：先删除 Customer 的更新视图，再更新 Order
// 如果 Order 外键引用 Customer → 约束违反
evm.save(em, customerUpdateView);  // 可能触发 DELETE
evm.save(em, orderUpdateView);     // 引用已删除的 customer

// ✅ 正确：先处理子实体，再处理父实体
// 或者将相关操作放在同一个 @Transactional 方法中，
// 在方法结束时统一 flush
evm.save(em, orderUpdateView);
evm.save(em, customerUpdateView);
em.flush();  // 统一刷新，Hibernate 自动排序 SQL

陷阱 3：缺少 @IdMapping

问题：声明 Entity View 时忘记 @IdMapping，启动时不一定报错，但运行时会出现奇怪的行为——去重失败、更新视图找不到目标行、分页结果错乱。 
// ❌ 错误：缺少 @IdMapping
@EntityView(Customer.class)
public interface CustomerBrokenView {
    Long getId();         // 只是普通 getter，不是 ID 映射！
    String getName();
}

// ✅ 正确：必须有 @IdMapping
@EntityView(Customer.class)
public interface CustomerCorrectView {
    @IdMapping            // 告诉 Blaze 这是主键字段
    Long getId();
    String getName();
}

// 启动时 Blaze 会校验 @UpdatableEntityView 必须有 @IdMapping
// 但只读 @EntityView 如果没有 @IdMapping，
// 只会在集合去重时出现意外行为

陷阱 4：循环 View 引用

问题：Customer View 包含 Order View 列表，Order View 又包含 Customer View——形成循环引用，Jackson 序列化时无限递归，最终 StackOverflowError。 
// ❌ 错误：循环引用
@EntityView(Customer.class)
public interface CustomerCircularView {
    @IdMapping Long getId();
    List<OrderCircularView> getOrders();
}

@EntityView(Order.class)
public interface OrderCircularView {
    @IdMapping Long getId();
    CustomerCircularView getCustomer();  // → 回到 Customer → 回到 Order → ∞
}

// ✅ 正确：打断循环——Order View 只取 Customer 的标量字段
@EntityView(Order.class)
public interface OrderSafeView {
    @IdMapping Long getId();
    @Mapping("customer.id")
    Long getCustomerId();        // 只取 ID
    @Mapping("customer.name")
    String getCustomerName();    // 只取 name
    // 不再引用完整的 CustomerView
}

本章小结

  • 项目架构采用 Controller → Service → Repository (EntityViewRepository) → EntityViewManager / CriteriaBuilderFactory → EntityManager → Hibernate → PostgreSQL 分层架构
  • Maven 依赖共需 7 个 Blaze-Persistence 模块：core-api/impl、entity-view-api/impl、integration-hibernate-6.2、integration-spring-data-3.3——版本必须一致
  • JPA Entity 定义了 5 个核心实体（Customer、Product、Order、OrderItem、Category），其中 Order 包含状态机和乐观锁，Category 支持自关联树形结构
  • BlazeConfig 只需 1 个 @Configuration 类，创建 CriteriaBuilderFactory 和 EntityViewManager 两个 Bean，配合 @EnableEntityViews 和 @EnableBlazeRepositories 完成自动装配
  • Entity View 投影体系为每个实体设计了 4 层 View（List/Detail/Create/Update），继承关系复用字段定义，运行时只查 View 声明的列——消除 SELECT *
  • Spring Data Repository 继承 EntityViewRepository 后，派生查询方法自动返回 Entity View，支持 Keyset 分页
  • Service 层 CRUD 使用 evm.save() 统一处理创建和更新，evm.find() 按 View 类型精确投影，evm.remove() 按 ID 删除
  • REST Controller 直接返回 Entity View，Jackson 自动序列化——无需额外 DTO 转换层
  • 全局异常处理通过 @RestControllerAdvice 统一拦截 5 类异常，返回一致的 ErrorResponse JSON
  • 4 类常见陷阱：LazyInitializationException（View 关联返回 Entity）、flush 顺序（外键依赖）、缺少 @IdMapping（去重失败）、循环 View 引用（StackOverflow）——每个都有明确的修复策略
